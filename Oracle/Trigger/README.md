```sql

CREATE OR REPLACE TRIGGER CHECK_SALAIO
BEFORE INSERT OR UPDATE ON ALUNO
FOR EACH ROW
BEGIN
		IF :NEW.SALARIO < 2000 THEN
				RAISE_APPLICATION_ERROR(-20000, 'VALOR INCORETO');
		END IF;
END;
/
```
